%    and by abstract/TOC pages.
%    We split this into two named styles:

%      'chapterstart' -> full mainpages header (first page of every chapter)
%
%    The trick: redefine 'plain' AFTER \pagestyle{mainpages} is set so
%    that \chapter's forced \thispagestyle{plain} still shows the header.
%    We do this by making 'plain' a copy of 'mainpages' from that point on.

  \fancyhf{}
  \setlength{\headheight}{2.2cm}
  \renewcommand{\headrulewidth}{0.4pt}

      Tudor-Alexandru Matei\\
      AirNest -- IoT Indoor Air Station}%
  }

    \raisebox{-0.55cm}{\includegraphics[height=1.55cm]{logo_upt}}%
  }
  \fancyfoot[C]{\fontsize{10}{12}\selectfont- \thepage\ -}
}
